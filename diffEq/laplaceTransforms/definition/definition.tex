\section{Definition}
\begin{definition}
	Let $f(t)$ be a function that is defined for $t \geq 0$.
	\begin{equation*}
		\Laplace{f} = \int_{0}^{\infty}{e^{-st}f(t) \mathrm{d}t}
	\end{equation*}
	The domain of $F$ is all values of $s$ where the integral is defined and finite.
\end{definition}
