\begin{example}
	Solve the following 1st order linear ODE.
	\begin{equation*}
		y^\prime - y = 2e^{x}
	\end{equation*}
\end{example}
$a(x) = -1$, so
\begin{equation*}
	\mu(x) = e^{\int{-1 \mathrm{d}x}} = e^{-x}
\end{equation*}
Applying to our equation and solving,
\begin{equation*}
	y^\prime e^{-x} - ye^{-x} = 2 \implies \left(ye^{-x}\right)^\prime = 2 \implies ye^{-x} = 2x + C \implies y = \frac{2x + C}{e^{-x}}
\end{equation*}
