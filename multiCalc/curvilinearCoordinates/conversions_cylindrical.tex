\subsection{Conversions}
\noindent
\begin{itemize}
	\item From Cylindrical to Cartesian
		\begin{equation*}
			(r, \theta, z) = (r\cos{\theta}, r\sin{\theta}, z)
		\end{equation*}
	\item From Cartesian to Cylindrical
		\begin{equation*}
			(x, y, z)=\left(\sqrt{x^2 + y^2}, \tan^{-1}{\frac{y}{x}}, z\right)
		\end{equation*}
\end{itemize}
