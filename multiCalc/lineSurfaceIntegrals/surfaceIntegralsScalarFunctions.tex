\subsection{Surface Integrals of Scalar Functions}
\begin{definition}
	The surface integral of a scalar function $f(x,y,z)$ is
	\begin{equation*}
		\iint\limits_{S}{f(x,y,z)\mathrm{d}s} = \iint\limits_{D}{(f\circ\vec{r})\norm{\vec{r_u} \times \vec{r_v}}\mathrm{d}A}
	\end{equation*}
\end{definition}
